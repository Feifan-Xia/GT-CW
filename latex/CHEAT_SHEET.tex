% 1. 段落结尾加 \\
Paragraph text here. \\

% 2. 用 \section 和 \subsection (不写数字)
\section{Main Topic}
\subsection{Sub Topic}

% 3. 窄图用 figure[H] + \columnwidth
\begin{figure}[H]
\centering
\includegraphics[width=\columnwidth]{image.png}
\caption{Description without "Figure X: " prefix}
\label{fig:label}
\end{figure}

% 4. 宽图用 figure*[t] + 0.8\textwidth
\begin{figure*}[t]
\centering
\includegraphics[width=0.8\textwidth]{wide_image.png}
\caption{wide figure description}
\label{fig:wide}
\end{figure*}

% 5. 表格用 \resizebox 自适应列宽
\begin{table}[H]
\centering
\small
\resizebox{\columnwidth}{!}{%
\begin{tabular}{lcc}
\toprule
... & ... & ... \\
\bottomrule
\end{tabular}%
}
\caption{table description (no "Table X:" prefix)}
\label{tab:label}
\end{table}

% 6. 列表不加 \\
\begin{itemize}
    \item First item
    \item Second item
\end{itemize}

% 7. 列表后的段落加 \\
Next paragraph after list. \\

% ============================================
% DON'Ts (❌ 不要这些)
% ============================================

% ❌ 不要在标题中写数字
\section{1. Introduction}  % WRONG!
\section{Introduction}     % CORRECT

% ❌ 不要在图题中写 "Figure X: "
\caption{Figure 1: My description}      % WRONG!
\caption{My description}                % CORRECT

% ❌ 不要在表题中写 "Table Y: "
\caption{Table 2: Results}              % WRONG!
\caption{Results}                       % CORRECT

% ❌ 不要忘记段尾 \\
This is a paragraph.
Next paragraph.           % WRONG - missing \\

This is a paragraph. \\
Next paragraph. \\        % CORRECT

% ❌ 不要多张双列图放同一页
\begin{figure*}[t]...\end{figure*}
\begin{figure*}[t]...\end{figure*}  % WRONG - max 1 per page!

% ❌ 不要用 figure[H] 放宽图
\begin{figure}[H]
\includegraphics[width=0.9\textwidth]{...}  % WRONG - use figure*[t]
\end{figure}

% ❌ 不要让表格溢出列外
\begin{tabular}{...}...  % WRONG - will overflow!
\resizebox{\columnwidth}{!}{\begin{tabular}{...}…}  % CORRECT

% ❌ 不要在列表项加 \\
\begin{itemize}
    \item Item one \\     % WRONG!
    \item Item two \\     % WRONG!
\end{itemize}
